CREATE SEQUENCE ACN START WITH 10000000 INCREMENT BY 1;

CREATE SEQUENCE USERID START WITH 20000000 INCREMENT BY 1;

CREATE SEQUENCE DEBIT START WITH 30000000 INCREMENT BY 1;

CREATE SEQUENCE TRANSID START WITH 40000000 INCREMENT BY 1;


CREATE TABLE ACCOUNT(
acno NUMBER(8) PRIMARY KEY NOT NULL,
CNAME VARCHAR2(20),
FATHER_NAME VARCHAR2(20),
DOB DATE,
MARITAL VARCHAR2(20),
BRANCH_NO NUMBER(7,2),
TYPE_OF_ACCOUNT VARCHAR2(20),
BALANCE NUMBER(10));


CREATE TABLE BADDRESS(
acno NUMBER(8) PRIMARY KEY NOT NULL REFERENCES ACCOUNT(acno) ON DELETE CASCADE,
CITY VARCHAR2(20),
STATE VARCHAR2(20),
COUNTRIES VARCHAR2(20),
PINCODE NUMBER(7));


CREATE TABLE BCONTACT(
acno NUMBER(8) REFERENCES ACCOUNT(ACNO) ON DELETE CASCADE,
PHONENO NUMBER(10),
EMAILID VARCHAR2(20)
);

CREATE TABLE BTRANSACTION(
transid NUMBER(8) PRIMARY KEY NOT NULL,
fromacc NUMBER(8) REFERENCES ACCOUNT(acno) ON DELETE CASCADE,
toacc NUMBER(8) REFERENCES ACCOUNT(acno) ON DELETE CASCADE,
AMOUNT NUMBER(10),
TDATE DATE);

CREATE TABLE BLOGIN(
USERID NUMBER(8) PRIMARY KEY NOT NULL,
NAME VARCHAR2(20),
PASSWORD VARCHAR2(20),
Q1 VARCHAR2(20),
Q2 VARCHAR2(20),
Q3 VARCHAR2(20),
A1 VARCHAR2(20),
A2 VARCHAR2(20),
A3 VARCHAR2(20));

create table map(
acno NUMBER(8)  NOT NULL REFERENCES ACCOUNT(acno) ON DELETE CASCADE,
userid NUMBER(8)  NOT NULL REFERENCES BLOGIN(USERID) ON DELETE CASCADE,
PRIMARY KEY(acno,userid)
);

create table joi(a1 NUMBER(8) REFERENCES ACCOUNT(acno) ON DELETE CASCADE,
a2 NUMBER(8) REFERENCES ACCOUNT(acno) ON DELETE CASCADE);
